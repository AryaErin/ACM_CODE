最短路变形
```cpp
#include<cstdio>
#include<iostream>
#include<cstring>
#include<algorithm>
#include<queue>
using namespace std;
const int INF = 0x3f3f3f3f;
const int MAXN = 1111;
int n,m,dist[MAXN];
bool vis[MAXN];
struct Graph{
    int tot, head[MAXN], to[MAXN*MAXN], weight[MAXN*MAXN], nxt[MAXN*MAXN];
    void Init(){ tot = 0; memset(head,255,sizeof(head)); }
    void ADDEDGE(int u, int v, int w){
        tot++;
        to[tot] = v;
        nxt[tot] = head[u];
        weight[tot] = w;
        head[u] = tot;
    }
}G;
void SPFA(){
    memset(dist,0,sizeof(dist));
    memset(vis,0,sizeof(vis));
    dist[1] = INF;
    queue<int> que;
    que.push(1);
    while(!que.empty()){
        int u = que.front();
        que.pop();
        vis[u] = false;
        for(int i = G.head[u]; ~i; i = G.nxt[i]){
            int v = G.to[i];
            int w = G.weight[i];
            if(dist[v]<min(dist[u],w)){
                dist[v] = min(dist[u],w);
                if(!vis[v]){
                    vis[v] = true;
                    que.push(v);
                }
            }
        }
    }
}
void solve(int kase){
    G.Init();
    scanf("%d %d",&n,&m);
    for(int i = 1; i <= m; i++){
        int u, v, w;
        scanf("%d %d %d",&u,&v,&w);
        G.ADDEDGE(u,v,w);
        G.ADDEDGE(v,u,w);
    }
    SPFA();
    printf("Scenario #%d:\n%d\n\n",kase,dist[n]);
}
int main(){
    int T;
    scanf("%d",&T);
    for(int kase = 1; kase <= T; kase++) solve(kase);
    return 0;
}
```
