拓扑排序
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<queue>
using namespace std;
const int MAXN = 111;
int n,ind[MAXN];
vector<int> vec[MAXN];
int main(){
    while(scanf("%d",&n)!=EOF){
        memset(ind,0,sizeof(ind));
        for(int i = 1; i <= n; i++){
            vec[i].clear();
            int v;
            while(scanf("%d",&v)&&v){
                vec[i].push_back(v);
                ind[v]++;
            }
        }
        queue<int> que;
        vector<int> res;
        for(int i = 1; i <= n; i++) if(!ind[i]) que.push(i);
        while(!que.empty()){
            int u = que.front();
            que.pop();
            res.push_back(u);
            for(int i = 0; i < (int)vec[u].size(); i++){
                if(--ind[vec[u][i]]==0) que.push(vec[u][i]);
            }
        }
        for(int i = 0; i < n; i++) printf("%d ",res[i]); puts("");
    }
    return 0;
}
```
