二分图匹配
```cpp
#include<cstring>
#include<cstdio>
#include<iostream>
#include<vector>
#include<cmath>
using namespace std;
const int MAXN = 111;
int n,m,s,v,match[MAXN],vis[MAXN];
vector<int> G[MAXN];
pair<double,double> ca[MAXN],cb[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int hungary(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
int main(){
    while(scanf("%d %d %d %d",&n,&m,&s,&v)!=EOF){
        for(int i = 0; i < MAXN; i++) G[i].clear();
        for(int i = 1; i <= n; i++) scanf("%lf %lf",&ca[i].first,&ca[i].second);
        for(int i = 1; i <= m; i++) scanf("%lf %lf",&cb[i].first,&cb[i].second);
        for(int i = 1; i <= n; i++) for(int j = 1; j <= m; j++){
            double dis = (ca[i].first-cb[j].first)*(ca[i].first-cb[j].first)+(ca[i].second-cb[j].second)*(ca[i].second-cb[j].second);
            if(v*v*s*s>=dis) G[i].push_back(j);
        }
        printf("%d\n",n-hungary());
    }
    return 0;
}
```
