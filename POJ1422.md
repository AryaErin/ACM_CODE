最小路径覆盖，现在把每个点拆分成两个，分别在二分图的左部和右部，图上的每一条边x->y就从左图的x
向右图的y连边，一开始相当于每个点是一条路径，如果匹配上两个点，那么总路径数量就-1，答案就是总点数-最大匹配
```cpp
#include<cstring>
#include<cstdio>
#include<iostream>
using namespace std;
const int MAXN = 222;
int match[MAXN];
bool vis[MAXN];
struct Graph{
    int tot,head[MAXN],to[MAXN<<2],nxt[MAXN<<2];
    void Init(){ tot = 0; memset(head,255,sizeof(head)); }
    void ADDEDGE(int u, int v){
        tot++;
        to[tot] = v;
        nxt[tot] = head[u];
        head[u] = tot;
    }
}G;
int T,n,m;
bool dfs(int u){
    vis[u] = true;
    for(int i = G.head[u]; ~i; i = G.nxt[i]){
        int v = G.to[i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int gao(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
void solve(){
    scanf("%d %d",&n,&m);
    G.Init();
    for(int i = 1; i <= m; i++){
        int u, v;
        scanf("%d %d",&u,&v);
        G.ADDEDGE(u,v);
    }
    printf("%d\n",n-gao());
}
int main(){
    for(scanf("%d",&T); T; T--) solve();
    return 0;
}
```
