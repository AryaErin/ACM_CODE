```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<queue>
#include<algorithm>
using namespace std;
const int MAXN = 222;
const int INF = 0x3f3f3f3f;
int n,m,k,rk[MAXN],iter[MAXN];
#define S 0
#define T MAXN - 1
struct EDGE{
    int to,cap,rev;
    EDGE(){}
    EDGE(int _to, int _cap, int _rev){ to = _to; cap = _cap; rev = _rev; }
};
vector<EDGE> G[MAXN];
vector<pair<int,pair<int,int> > > vec;
void ADDEDGE(int u, int v, int cap){
    G[u].push_back(EDGE(v,cap,(int)G[v].size()));
    G[v].push_back(EDGE(u,0,(int)G[u].size()-1));
}
bool bfs(){
    memset(rk,0,sizeof(rk));
    memset(iter,0,sizeof(iter));
    queue<int> que;
    que.push(S);
    rk[S] = 1;
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)G[u].size(); i++){
            EDGE &e = G[u][i];
            if(!e.cap || rk[e.to]) continue;
            rk[e.to] = rk[u] + 1;
            que.push(e.to);
        }
    }
    return rk[T]!=0;
}
int dfs(int u, int f){
    if(u==T) return f;
    for(int &i = iter[u]; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(!e.cap || rk[e.to]!=rk[u] + 1) continue;
        int d = dfs(e.to,min(f,e.cap));
        if(d){
            e.cap -= d;
            G[e.to][e.rev].cap += d;
            return d;
        }
    }
    return 0;
}
int Dinic(int mid){
    for(int i = 0; i <= n; i++) G[i].clear();
    G[T].clear();
    for(int i = 0; i < (int)vec.size(); i++){
        if(vec[i].first>mid) break;
        ADDEDGE(vec[i].second.first,vec[i].second.second,1);
        ADDEDGE(vec[i].second.second,vec[i].second.first,1);
    }
    ADDEDGE(S,1,k);
    ADDEDGE(n,T,k);
    int flow = 0;
    while(bfs()){
        int d = dfs(S,INF);
        while(d){
            flow += d;
            d = dfs(S,INF);
        }
    }
    return flow;
}
int main(){
    while(scanf("%d %d %d",&n,&m,&k)!=EOF){
        vec.clear();
        int l = 1, r = 0;
        for(int i = 1; i <= m; i++){
            int u, v, c;
            scanf("%d %d %d",&u,&v,&c);
            vec.push_back(make_pair(c,make_pair(u,v)));
            r = max(r,c);
        }
        sort(vec.begin(),vec.end());
        while(l<=r){
            int mid = (l+r) >> 1;
            if(Dinic(mid)==k) r = mid - 1;
            else l = mid + 1;
        }
        printf("%d\n",l);
    }
    return 0;
}
```
