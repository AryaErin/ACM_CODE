强连通分量分解之后找出度为0的强连通分量里的点，按顺序输出
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
const int MAXN = 5555;
int n,m,dfn[MAXN],low[MAXN],idx,num,instk[MAXN],sccid[MAXN],deg[MAXN];
vector<int> G[MAXN];
stack<int> stk;
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    stk.push(u);
    instk[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(low[u]==dfn[u]){
        num++;
        int tp;
        do{
            tp = stk.top();
            stk.pop();
            sccid[tp] = num;
            instk[tp] = false;
        }while(tp!=u);
    }
}
void rebuild(){
    for(int i = 1; i <= num; i++) deg[i] = 0;
    for(int u = 1; u <= n; u++){
        for(int j = 0; j < (int)G[u].size(); j++){
            int v = G[u][j];
            if(sccid[u]==sccid[v]) continue;
            deg[sccid[u]]++;
        }
    }
}
int main(){
    while(scanf("%d",&n)!=EOF && n){
        scanf("%d",&m);
        for(int i = 1; i <= n; i++){
            G[i].clear();
            low[i] = dfn[i] = instk[i] = 0;
        }
        idx = num = 0;
        for(int i = 1; i <= m; i++){
            int u, v;
            scanf("%d %d",&u,&v);
            G[u].push_back(v);
        }
        for(int i = 1; i <= n; i++) if(!dfn[i]) tarjan(i);
        rebuild();
        vector<int> vec;
        for(int i = 1; i <= n; i++) if(!deg[sccid[i]]) vec.push_back(i);
        sort(vec.begin(),vec.end());
        for(int i = 0; i < (int)vec.size(); i++) printf("%d ",vec[i]); puts("");
    }
    return 0;
}
```
