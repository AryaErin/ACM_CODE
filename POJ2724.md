所有二进制编码只差1位奶酪的进行连边，然后跑最大匹配，用n-最大匹配即可
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
const int MAXN = 2222;
int n,m,vis[MAXN],match[MAXN];
set<int> S;
char buf[MAXN];
vector<int> G[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int hungary(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF && n + m){
        S.clear();
        for(int i = 0; i < MAXN; i++) G[i].clear();
        for(int i = 1; i <= m; i++){
            scanf("%s",buf);
            int numa = 0, numb = 0;
            for(int bit = 0; bit < n; bit++){
                if(isdigit(buf[bit])) numa ^= (1 << bit) * (buf[bit]-'0'), numb ^= (1 << bit) *(buf[bit]-'0');
                else numa ^= (1 << bit);
            }
            S.insert(numa); S.insert(numb);
        }
        vector<int> vec;
        for(set<int>::iterator it = S.begin(); it != S.end(); it++) vec.push_back(*it);
        for(int i = 0; i < (int)vec.size(); i++) for(int j = 0; j < (int)vec.size(); j++){
            if(__builtin_popcount(vec[i]^vec[j])==1) G[i+1].push_back(j+1);
        }
        n = vec.size();
        printf("%d\n",n - hungary()/2);
    }
    return 0;
}
```
