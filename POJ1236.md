先SCC缩点，对于第一个问题，只有那些入度为0的点才是必须要software的，输出入度为0的点即可。对于第二个问题，问最少要加几条边能使得图变成强连通图，取入度为0
的点和出度为0的点的个数的最大值即可，这样保证每个点都有出入度，都能从其他联通分量来，到其他连通分量去。
```cpp
#include<iostream>
#include<cstdio>
#include<cstring>
#include<algorithm>
#include<vector>
#include<stack>
using namespace std;
const int MAXN = 111;
vector<int> G[MAXN];
int n,dfn[MAXN],low[MAXN],num,idx,sccid[MAXN],indeg[MAXN],outdeg[MAXN];
bool instk[MAXN];
stack<int> stk;
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    stk.push(u);
    instk[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(low[u]==dfn[u]){
        num++;
        int tp;
        do{
            tp = stk.top();
            stk.pop();
            sccid[tp] = num;
            instk[tp] = false;
        }while(tp!=u);
    }
}
void rebuild(){
    for(int u = 1; u <= n; u++){
        for(int i = 0; i < (int)G[u].size(); i++){
            int v = G[u][i];
            if(sccid[u]==sccid[v]) continue;
            indeg[sccid[v]]++;
            outdeg[sccid[u]]++;
        }
    }
}
int main(){
    scanf("%d",&n);
    for(int i = 1; i <= n; i++){
        int v;
        while(scanf("%d",&v),v) G[i].push_back(v);
    }
    for(int i = 1; i <= n; i++) if(!dfn[i]) tarjan(i);
    if(num==1) return printf("1\n0\n"), 0;
    rebuild();
    int zeroindeg = 0;
    for(int i = 1; i <= num; i++) if(!indeg[i]) zeroindeg++;
    int nind = 0, noutd = 0;
    for(int i = 1; i <= num; i++){
        if(!indeg[i]) nind++;
        if(!outdeg[i]) noutd++;
    }
    printf("%d\n%d\n",zeroindeg,max(nind,noutd));
    return 0;
}
```
