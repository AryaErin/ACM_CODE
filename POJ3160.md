所有负权点变成0，然后缩点DP找DAG最长路

#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 3e4+7;
int n,m,dfn[MAXN],low[MAXN],instk[MAXN],num,idx,sccid[MAXN],val[MAXN],sccv[MAXN],deg[MAXN],f[MAXN];
vector<int> G[MAXN],NG[MAXN];
stack<int> stk;
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    instk[u] = 1;
    stk.push(u);
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(low[u]==dfn[u]){
        num++;
        int tp;
        sccv[num] = 0;
        do{
            tp = stk.top();
            stk.pop();
            sccid[tp] = num;
            sccv[num] += val[tp];
            instk[tp] = 0;
        }while(tp!=u);
    }
}
void rebuild(){
    for(int u = 1; u <= n; u++){
        for(int i = 0; i < (int)G[u].size(); i++){
            int v = G[u][i];
            if(sccid[u]==sccid[v]) continue;
            NG[sccid[v]].push_back(sccid[u]);
            deg[sccid[u]]++;
        }
    }
}
int solve(){
    queue<int> que;
    for(int i = 1; i <= num; i++) if(!deg[i]){
        f[i] = sccv[i];
        que.push(i);
    }
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)NG[u].size(); i++){
            int v = NG[u][i];
            f[v] = max(f[v],f[u]+sccv[v]);
            if(!--deg[v]) que.push(v);
        }
    }
    return *max_element(f+1,f+1+num);
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        for(int i = 1; i <= n; i++){
            dfn[i] = low[i] = instk[i] = deg[i] = f[i] = 0;
            G[i].clear(); NG[i].clear();
        }
        num = idx = 0;
        for(int i = 1; i <= n; i++){
            scanf("%d",&val[i]);
            if(val[i]<0) val[i] = 0;
        }
        for(int i = 1; i <= m; i++){
            int u, v;
            scanf("%d %d",&u,&v);
            u++, v++;
            G[u].push_back(v);
        }
        for(int i = 1; i <= n; i++) if(!dfn[i]) tarjan(i);
        rebuild();
        printf("%d\n",solve());
    }
    return 0;
}
