边双连通分量分解，缩点之后变成一棵树，答案就是度数为0的点的一半向上取整
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 5555;
int n,m,deg[MAXN],dfn[MAXN],low[MAXN],num,idx,bccid[MAXN];
vector<int> G[MAXN];
stack<int> stk;
void tarjan(int u, int f){
    dfn[u] = low[u] = ++idx;
    stk.push(u);
    bool tag = false;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v, u);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(v==f&&!tag){
                tag = true;
                continue;
            }
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(low[u]==dfn[u]){
        num++;
        int tp;
        do{
            tp = stk.top();
            stk.pop();
            bccid[tp] = num;
        }while(tp!=u);
    }
}
void rebuild(){
    for(int i = 1; i <= num; i++) deg[i] = 0;
    for(int u = 1; u <= n; u++){
        for(int i = 0; i < (int)G[u].size(); i++){
            int v = G[u][i];
            if(bccid[u]==bccid[v]) continue;
            deg[bccid[u]]++; deg[bccid[v]]++;
        }
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        for(int i = 1; i <= n; i++) G[i].clear();
        for(int i = 1; i <= m; i++){
            int u, v;
            scanf("%d %d",&u,&v);
            G[u].push_back(v);
            G[v].push_back(u);
        }
        memset(dfn,0,sizeof(dfn));
        memset(low,0,sizeof(low));
        num = idx = 0;
        tarjan(1,0);
        rebuild();
        int tot = 0;
        for(int i = 1; i <= num; i++) if(deg[i]==2) tot++;
        printf("%d\n",tot-tot/2);
    }
    return 0;
}
```
