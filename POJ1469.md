二分图匹配，如果所有课程都有匹配的话就是YES，否则就是NO
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
using namespace std;
const int MAXN = 555;
int T,n,m,match[MAXN];
bool vis[MAXN];
vector<int> G[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
bool hungary(){
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(!dfs(i)) return false;
    }
    return true;
}
void solve(){
    scanf("%d %d",&n,&m);
    for(int i = 1; i <= n+m; i++) G[i].clear();
    for(int i = 1; i <= n; i++){
        int num;
        scanf("%d",&num);
        while(num--){
            int v;
            scanf("%d",&v);
            G[i].push_back(v);
        }
    }
    if(hungary()) puts("YES");
    else puts("NO");
}
int main(){
    for(scanf("%d",&T); T; T--) solve();
    return 0;
}
```
