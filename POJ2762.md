SCC缩点+topo排序

#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 111111;
int n,m,dfn[MAXN],low[MAXN],instk[MAXN],num,idx,sccid[MAXN],ind[MAXN];
stack<int> stk;
vector<int> G[MAXN],nG[MAXN];
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    instk[u] = 1;
    stk.push(u);
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(dfn[u]==low[u]){
        num++;
        int tp;
        do{
            tp = stk.top();
            stk.pop();
            instk[tp] = 0;
            sccid[tp] = num;
        }while(tp!=u);
    }
}
void rebuild(){
    for(int i = 1; i <= num; i++) nG[i].clear();
    for(int i = 1; i <= num; i++) ind[i] = 0;
    for(int u = 1; u <= n; u++){
        for(int i = 0; i < (int)G[u].size(); i++){
            int v = G[u][i];
            if(sccid[u]==sccid[v]) continue;
            nG[sccid[u]].push_back(sccid[v]);
            ind[sccid[v]]++;
        }
    }
}
bool topo(){
    queue<int> que;
    for(int i = 1; i <= num; i++) if(!ind[i]) que.push(i);
    while(!que.empty()){
        if(que.size()>1) return false;
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)nG[u].size(); i++){
            int v = nG[u][i];
            if(!--ind[v]) que.push(v);
        }
    }
    return true;
}
void solve(){
    scanf("%d %d",&n,&m);
    for(int i = 1; i <= n; i++){
        G[i].clear();
        dfn[i] = low[i] = instk[i] = 0;
    }
    num = idx = 0;
    for(int i = 1; i <= m; i++){
        int u, v;
        scanf("%d %d",&u,&v);
        G[u].push_back(v);
    }
    for(int i = 1; i <= n; i++) if(!dfn[i]) tarjan(i);
    rebuild();
    if(topo()) puts("Yes");
    else puts("No");
}
int main(){
    int T;
    for(scanf("%d",&T); T; T--) solve();
    return 0;
}
