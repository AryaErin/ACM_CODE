枚举右边界，然后二分长度网络流，取最小区间长度

#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 1111;
const int INF = 0x3f3f3f3f;
int n, m, rk[MAXN], iter[MAXN];
#define S 0
#define T MAXN - 1
vector<int> vec[MAXN];
struct EDGE{
    int to,cap,rev;
    EDGE(){}
    EDGE(int _to, int _cap, int _rev){ to = _to; cap = _cap; rev = _rev; }
};
vector<EDGE> G[MAXN];
void ADDEDGE(int u, int v, int cap){
    G[u].push_back(EDGE(v,cap,(int)G[v].size()));
    G[v].push_back(EDGE(u,0,(int)G[u].size()-1));
}
bool bfs(){
    memset(rk,0,sizeof(rk));
    memset(iter,0,sizeof(iter));
    queue<int> que;
    que.push(S);
    rk[S] = 1;
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)G[u].size(); i++){
            EDGE &e = G[u][i];
            if(!e.cap || rk[e.to]) continue;
            rk[e.to] = rk[u] + 1;
            que.push(e.to);
        }
    }
    return rk[T]!=0;
}
int dfs(int u, int f){
    if(u==T) return f;
    for(int &i = iter[u]; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(!e.cap || rk[e.to]!=rk[u] + 1) continue;
        int d = dfs(e.to,min(f,e.cap));
        if(d){
            e.cap -= d;
            G[e.to][e.rev].cap += d;
            return d;
        }
    }
    return 0;
}
int Dinic(){
    int flow = 0;
    while(bfs()){
        int d = dfs(S,INF);
        while(d){
            flow += d;
            d = dfs(S,INF);
        }
    }
    return flow;
}
int prefer[MAXN][22],cp[MAXN];
bool check(int mid, int st){
    for(int i = 0; i <= n + m; i++) G[i].clear();
    G[T].clear();
    for(int i = 1; i <= n; i++){
        ADDEDGE(S,i,1);
        for(int j = st - mid + 1; j <= st; j++) ADDEDGE(i,prefer[i][j]+n,1);
    }
    for(int i = 1; i <= m; i++) ADDEDGE(i+n,T,cp[i]);
    return Dinic() == n;
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        for(int i = 1; i <= n; i++) for(int j = 1; j <= m; j++) scanf("%d",&prefer[i][j]);
        for(int i = 1; i <= m; i++) scanf("%d",&cp[i]);
        int minn = m;
        for(int i = 1; i <= m; i++){
            int l = 1, r = i;
            while(l<=r){
                int mid = (l+r) >> 1;
                if(check(mid,i)) r = mid - 1;
                else l = mid + 1;
            }
            if(l==i+1) continue;
            else minn = min(minn,l);
        }
        printf("%d\n",minn);
    }
    return 0;
}
