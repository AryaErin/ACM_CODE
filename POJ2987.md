最大权闭合子图模板题

S向正权点连边，边权为权值；负权点向T连边，边权为权值的绝对值

由继承关系的点按继承关系连边，边权为INF

找出最小割，最大权值为所有正点权的和-最小割

从源点开始DFS，删去的点为DFS到的点数-1（S不算）

```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 5555;
const int INF = 0x3f3f3f3f;
#define S 0
#define T MAXN - 1
int n,m,cost[MAXN],rk[MAXN],iter[MAXN],vis[MAXN];
struct EDGE{
    int to, cap, rev;
    EDGE(){}
    EDGE(int _to, int _cap, int _rev){ to = _to; cap = _cap; rev = _rev; } 
};
vector<EDGE> G[MAXN];
void ADDEDGE(int u, int v, int cap){
    G[u].push_back(EDGE(v,cap,(int)G[v].size()));
    G[v].push_back(EDGE(u,0,(int)G[u].size()-1));
}
bool bfs(){
    memset(iter,0,sizeof(iter));
    memset(rk,0,sizeof(rk));
    rk[S] = 1;
    queue<int> que;
    que.push(S);
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)G[u].size(); i++){
            EDGE &e = G[u][i];
            if(!e.cap || rk[e.to]) continue;
            rk[e.to] = rk[u] + 1;
            que.push(e.to);
        }
    }
    return rk[T] != 0;
}
int dfs(int u, int f){
    if(u==T) return f;
    for(int &i = iter[u]; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(!e.cap || rk[e.to] != rk[u] + 1) continue;
        int d = dfs(e.to,min(f,e.cap));
        if(d){
            e.cap -= d;
            G[e.to][e.rev].cap += d;
            return d;
        }
    }
    return 0;
}
long long Dinic(){
    long long flow = 0;
    while(bfs()){
        int d = dfs(S,INF);
        while(d){
            flow += d;
            d = dfs(S,INF);
        }
    }
    return flow;
}
void search(int u, int &res){
    vis[u] = true;
    res++;
    for(int i = 0; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(e.cap&&!vis[e.to]) search(e.to,res); 
    }
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        long long tot = 0;
        for(int i = 0; i <= n; i++) G[i].clear();
        G[T].clear();
        for(int i = 1; i <= n; i++){
            scanf("%d",&cost[i]);
            if(cost[i]>0){
                ADDEDGE(S,i,cost[i]);
                tot += cost[i];
            }
            else ADDEDGE(i,T,-cost[i]);
        }
        for(int i = 1; i <= m; i++){
            int u, v;
            scanf("%d %d",&u,&v);
            ADDEDGE(u,v,INF);
        }
        long long flow = Dinic();
        memset(vis,0,sizeof(vis));
        int res = -1;
        search(S,res);
        printf("%d %lld\n",res,tot-flow);
    }
    return 0;
}
```
