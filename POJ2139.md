#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int MAXN = 333;
int n,m,dist[MAXN][MAXN],buf[MAXN];
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        memset(dist,0x3f,sizeof(dist));
        for(int i = 1; i <= m; i++){
            int k; scanf("%d",&k);
            for(int j = 1; j <= k; j++) scanf("%d",&buf[j]);
            for(int I = 1; I < k; I++) for(int J = I + 1; J <= k; J++) dist[buf[I]][buf[J]] = dist[buf[J]][buf[I]] = 1;
        }
        for(int i = 1; i <= n; i++) dist[i][i] = 0;
        for(int k = 1; k <= n; k++) for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++) dist[i][j] = min(dist[i][j],dist[i][k]+dist[k][j]);
        int minn = 0x3f3f3f3f;
        for(int i = 1; i <= n; i++){
            int tot = 0;
            for(int j = 1; j <= n; j++) tot += dist[i][j];
            tot = tot * 100 / (n-1);
            if(tot<minn) minn = tot;
        }
        printf("%d\n",minn);
    }
    return 0;
}
