差分约束SPFA找最短路

#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 1e3+7;
const int INF = 0x3f3f3f3f;
int n,N,M,vis[MAXN],dist[MAXN],inq[MAXN];
vector<pair<int,int> > G[MAXN];
int spfa(){
    memset(inq,0,sizeof(inq));
    memset(dist,0,sizeof(dist));
    queue<int> que;
    for(int i = 1; i <= n; i++) {
        que.push(i);
        vis[i] = 1;
    }
    while(!que.empty()){
        int u = que.front();
        que.pop();
        vis[u] = 0;
        for(int i = 0; i < (int)G[u].size(); i++){
            pair<int,int> p = G[u][i];
            int v = p.first, d = p.second;
            if(dist[v]>dist[u]+d){
                dist[v] = dist[u] + d;
                if(!vis[v]){
                    vis[v] = true;
                    if(++inq[v]>=n) return -1;
                    que.push(v);
                }
            }
        }
    }
    memset(dist,INF,sizeof(dist));
    memset(vis,0,sizeof(vis));
    dist[1] = 0;
    que.push(1);
    while(!que.empty()){
        int u = que.front();
        que.pop();
        vis[u] = 0;
        for(int i = 0; i < (int)G[u].size(); i++){
            pair<int,int> p = G[u][i];
            int v = p.first, d = p.second;
            if(dist[v]>dist[u]+d){
                dist[v] = dist[u] + d;
                if(!vis[v]){
                    vis[v] = 1;
                    que.push(v);
                }
            }
        }
    }
    if(dist[n]==INF) return -2;
    return dist[n];
}
int main(){
    while(scanf("%d %d %d",&n,&N,&M)!=EOF){
        for(int i = 1; i <= n; i++) G[i].clear();
        for(int i = 1; i < n; i++) G[i+1].push_back(make_pair(i,0));
        for(int i = 1; i <= N; i++){
            int u, v, c;
            scanf("%d %d %d",&u,&v,&c);
            if(u>v) u^=v^=u^=v;
            G[u].push_back(make_pair(v,c));
        }
        for(int i = 1; i <= M; i++){
            int u, v, c;
            scanf("%d %d %d",&u,&v,&c);
            if(u>v) u^=v^=u^=v;
            G[v].push_back(make_pair(u,-c));
        }
        printf("%d\n",spfa());
    }
    return 0;
}
