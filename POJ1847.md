最短路
往第一个连边边权为0，其余为1
```cpp
#include<cstring>
#include<cstdio>
#include<iostream>
using namespace std;
const int MAXN = 111;
int G[MAXN][MAXN];
int n,s,t;
int main(){
    scanf("%d %d %d",&n,&s,&t);
    memset(G,0x3f,sizeof(G));
    for(int i = 1; i <= n; i++){
        G[i][i] = 0;
        int k;
        scanf("%d",&k);    
        for(int j = 1; j <= k; j++){
            int v; scanf("%d",&v);
            if(j==1) G[i][v] = 0;
            else G[i][v] = 1;
        }
    }
    for(int k = 1; k <= n; k++) for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++){
        G[i][j] = min(G[i][j],G[i][k]+G[k][j]);
    }
    if(G[s][t]==0x3f3f3f3f) puts("-1");
    else printf("%d\n",G[s][t]);
    return 0;
}
```
