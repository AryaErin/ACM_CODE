#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 444;
const int INF = 0x3f3f3f3f;
int n,m,rk[MAXN],iter[MAXN],initG[MAXN][MAXN],mine[MAXN],cap[MAXN],GG[MAXN][MAXN];
#define S 0
#define T MAXN - 1
vector<int> vec[MAXN];
struct EDGE{
    int to,cap,rev;
    EDGE(){}
    EDGE(int _to, int _cap, int _rev){ to = _to; cap = _cap; rev = _rev; }
};
vector<EDGE> G[MAXN];
void ADDEDGE(int u, int v, int cap){
    G[u].push_back(EDGE(v,cap,(int)G[v].size()));
    G[v].push_back(EDGE(u,0,(int)G[u].size()-1));
}
bool bfs(){
    memset(rk,0,sizeof(rk));
    memset(iter,0,sizeof(iter));
    queue<int> que;
    que.push(S);
    rk[S] = 1;
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)G[u].size(); i++){
            EDGE &e = G[u][i];
            if(!e.cap || rk[e.to]) continue;
            rk[e.to] = rk[u] + 1;
            que.push(e.to);
        }
    }
    return rk[T]!=0;
}
int dfs(int u, int f){
    if(u==T) return f;
    for(int &i = iter[u]; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(!e.cap || rk[e.to]!=rk[u] + 1) continue;
        int d = dfs(e.to,min(f,e.cap));
        if(d){
            e.cap -= d;
            G[e.to][e.rev].cap += d;
            return d;
        }
    }
    return 0;
}
void build(int mid){
    for(int i = 0; i <= n*2; i++) G[i].clear();
    G[T].clear();
    memset(GG,INF,sizeof(GG));
    for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++) if(initG[i][j]<=mid) GG[i][j] = initG[i][j];
    for(int k = 1; k <= n; k++) for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++)
        GG[i][j] = min(GG[i][j],GG[i][k]+GG[k][j]);
    for(int i = 1; i <= n; i++) if(mine[i]) ADDEDGE(S,i,mine[i]);
    for(int i = 1; i <= n; i++) if(cap[i]) ADDEDGE(i+n,T,cap[i]);
    for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++) if(GG[i][j]!=INF) ADDEDGE(i,j+n,INF);
}
int Dinic(int mid){
    build(mid);
    int flow = 0;
    while(bfs()){
        int d = dfs(S,INF);
        while(d){
            flow += d;
            d = dfs(S,INF);
        }
    }
    return flow;
}
int main(){
    while(scanf("%d",&n)!=EOF&&n){
        int tot = 0;
        for(int i = 1; i <= n; i++){
            scanf("%d",&mine[i]);
            tot += mine[i];
        }
        for(int i = 1; i <= n; i++) scanf("%d",&cap[i]);
        memset(initG,INF,sizeof(initG));
        int maxe = 0;
        scanf("%d",&m);
        for(int i = 1; i <= m; i++) {
            int u, v, c;
            scanf("%d %d %d",&u,&v,&c);
            initG[u][v] = initG[v][u] = min(initG[u][v],c);
            maxe = max(maxe,c);
        }
        for(int i = 1; i <= n; i++) initG[i][i] = 0;
        int l = 1, r = maxe;
        while(l<=r){
            int mid = (l+r) >> 1;
            if(Dinic(mid)==tot) r = mid - 1;
            else l = mid + 1;
        }
        if(l>maxe) puts("No Solution");
        else printf("%d\n",l);
    }
    return 0;
}
