欧拉回路，每条路建正反两条边，题目保证有答案
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
using namespace std;
const int MAXN = 1e4+7;

struct Graph{
    int tot,head[MAXN],to[MAXN<<5],nxt[MAXN<<5];
    void clear(){ tot = 0; memset(head,255,sizeof(head)); }
    void ADDEDGE(int u, int v){
        to[tot] = v;
        nxt[tot] = head[u];
        head[u] = tot;
        tot++;
    }
}G;
int n,m,stk[MAXN<<5],top,vis[MAXN<<5];
void dfs(int u){
    for(int i = G.head[u]; ~i; i = G.nxt[i]){
        int v = G.to[i];
        if(vis[i]) continue;
        vis[i] = true;
        dfs(v);
    }
    stk[++top] = u;
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF){
        G.clear();
        top = 0;
        memset(vis,0,sizeof(vis));
        for(int i = 1; i <= m; i++){
            int u, v;
            scanf("%d %d",&u,&v);
            G.ADDEDGE(u,v);
            G.ADDEDGE(v,u);
        }
        dfs(1);
        while(top) printf("%d\n",stk[top--]);
    }
    return 0;
}
```
