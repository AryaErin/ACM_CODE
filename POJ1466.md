二分图的最大独立集 = 顶点数 - 最大匹配数
因为关系时相互的，所以答案是顶点数-最大匹配数的一半
```cpp
#include<cstdio>
#include<iostream>
#include<cstring>
#include<vector>
using namespace std;
const int MAXN = 555;
int match[MAXN],n;
bool vis[MAXN];
vector<int> G[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int solve(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
int main(){
    while(scanf("%d",&n)!=EOF){
        for(int i = 1; i <= n; i++) G[i].clear();
        for(int i = 1; i <= n; i++){
            int ID,num;
            scanf("%d: (%d)",&ID,&num);
            while(num--){
                int v;
                scanf("%d",&v);
                G[ID+1].push_back(v+1);
            }
        }
        printf("%d\n",n-solve()/2);
    }
    return 0;
}
```
