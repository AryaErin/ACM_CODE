输入有点恶心，其他就是简单构图跑网络流
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<queue>
using namespace std;
const int MAXN = 222;
const int INF = 0x3f3f3f3f;
#define S 0
#define T MAXN-1
int n,np,nc,m,rk[MAXN],iter[MAXN];
char buf[MAXN];
struct EDGE{
    int to, cap, rev;
    EDGE(){}
    EDGE(int _to, int _cap, int _rev){ to = _to; cap = _cap; rev = _rev; }
};
vector<EDGE> G[MAXN];
void ADDEDGE(int u, int v, int cap){
    G[u].push_back(EDGE(v,cap,(int)G[v].size()));
    G[v].push_back(EDGE(u,0,(int)G[u].size()-1));
}
bool bfs(){
    memset(iter,0,sizeof(iter));
    memset(rk,0,sizeof(rk));
    rk[S] = 1;
    queue<int> que;
    que.push(S);
    while(!que.empty()){
        int u = que.front();
        que.pop();
        for(int i = 0; i < (int)G[u].size(); i++){
            EDGE &e = G[u][i];
            if(!e.cap || rk[e.to]) continue;
            rk[e.to] = rk[u] + 1;
            que.push(e.to);
        }
    }
    return rk[T]!=0;
}
int dfs(int u, int f){
    if(u==T) return f;
    for(int &i = iter[u]; i < (int)G[u].size(); i++){
        EDGE &e = G[u][i];
        if(!e.cap || rk[e.to]!=rk[u]+1) continue;
        int d = dfs(e.to,min(f,e.cap));
        if(d){
            e.cap -= d;
            G[e.to][e.rev].cap += d;
            return d;
        }
    }
    return 0;
}
int Dinic(){
    int flow = 0;
    while(bfs()){
        int d = dfs(S,INF);
        while(d){
            flow += d;
            d = dfs(S,INF);
        }
    }
    return flow;
}
int main(){
    while(scanf("%d %d %d %d",&n,&np,&nc,&m)!=EOF){
        for(int i = 0; i < MAXN; i++) G[i].clear();
        for(int i = 1; i <= m; i++){
            scanf("%s",buf);
            int u, v, c;
            sscanf(buf,"(%d,%d)%d",&u,&v,&c);
            ADDEDGE(u+1,v+1,c);
        }
        for(int i = 1; i <= np; i++){
            int u, p;
            scanf("%s",buf);
            sscanf(buf,"(%d)%d",&u,&p);
            ADDEDGE(S,u+1,p);
        }
        for(int i = np + 1; i <= np + nc; i++){
            int u, c;
            scanf("%s",buf);
            sscanf(buf,"(%d)%d",&u,&c);
            ADDEDGE(u+1,T,c);
        }
        printf("%d\n",Dinic());
    }
    return 0;
}
```
