二分图匹配，主要考虑建边，用Floyd传递闭包，找出哪些类型的插头可以进行转换，然后左边点和右边点连边的时候，只要是可以转换的就连边，然后跑二分图匹配就完了
```cpp
#include<cstdio>
#include<iostream>
#include<cstring>
#include<vector>
#include<string>
#include<map>
using namespace std;
const int MAXN = 1111;
int n,m,k;
map<string,int> msk;
int exc[MAXN][MAXN];
int LX[MAXN],RX[MAXN],match[MAXN];
bool vis[MAXN];
vector<int> G[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int solve(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(0); cout.tie(0);
    int tag = 0;
    cin >> m;
    for(int i = 1; i <= m; i++){
        string sta;
        cin >> sta;
        if(!msk.count(sta)) msk.insert(make_pair(sta,++tag));
        RX[i] = msk.at(sta);
    }
    cin >> n;
    for(int i = 1; i <= n; i++){
        string sta;
        cin >> sta >> sta;
        if(!msk.count(sta)) msk.insert(make_pair(sta,++tag));
        LX[i] = msk.at(sta);
    }
    cin >> k;
    for(int i = 1; i <= k; i++){
        string sa,sb;
        cin >> sa >> sb;
        if(!msk.count(sa)) msk.insert(make_pair(sa,++tag));
        if(!msk.count(sb)) msk.insert(make_pair(sb,++tag));
        exc[msk.at(sa)][msk.at(sb)] = 1;
    }
    for(int i = 1; i <= tag; i++) exc[i][i] = 1;
    for(int kk = 1; kk <= tag; kk++) for(int i = 1; i <= tag; i++) for(int j = 1; j <= tag; j++){
        if(exc[i][kk]&&exc[kk][j]) exc[i][j] = 1;
    }
    for(int i = 1; i <= n; i++) for(int j = 1; j <= m; j++){
        if(!exc[LX[i]][RX[j]]) continue;
        G[i].push_back(j);
    }
    cout << n-solve() << endl;
    return 0;
}
```
