2sat，首先二分最远距离，然后根据限制条件建边判断是否可行即可
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<cmath>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 1111;
int n,N,M,dfn[MAXN],low[MAXN],instk[MAXN],sccid[MAXN],idx,num;
pair<int,int> s1,s2,cow[MAXN],frd[MAXN],emmm[MAXN];
vector<int> G[MAXN];
stack<int> stk;
int dist(pair<int,int> &A, pair<int,int> &B){ return abs(A.first-B.first) + abs(A.second-B.second); }
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    instk[u] = 1;
    stk.push(u);
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(dfn[u]==low[u]){
        num++;
        int tp;
        do{
            tp = stk.top();
            stk.pop();
            instk[tp] = 0;
            sccid[tp] = num;
        }while(tp!=u);
    }
}
bool _2sat(int m){
    for(int i = 1; i <= (n << 1); i++){
        G[i].clear();
        dfn[i] = low[i] = instk[i] = 0;
    }
    idx = num = 0;
    for(int i = 1; i <= N; i++){
        int u = emmm[i].first;
        int v = emmm[i].second;
        G[u].push_back(v+n);
        G[v].push_back(u+n);
        G[u+n].push_back(v);
        G[v+n].push_back(u);
    }
    for(int i = 1; i <= M; i++){
        int u = frd[i].first;
        int v = frd[i].second;
        G[u].push_back(v);
        G[u+n].push_back(v+n);
        G[v].push_back(u);
        G[v+n].push_back(u+n);
    }
    for(int i = 1; i <= n; i++) for(int j = i + 1; j <= n; j++){
        if(dist(cow[i],s1)+dist(cow[j],s1)>m){
            G[i].push_back(j+n);
            G[j].push_back(i+n);
        }
        if(dist(cow[i],s2)+dist(cow[j],s2)>m){
            G[i+n].push_back(j);
            G[j+n].push_back(i);
        }
        if(dist(cow[i],s1)+dist(cow[j],s2)+dist(s1,s2)>m){
            G[i].push_back(j);
            G[j+n].push_back(i+n);
        }
        if(dist(cow[i],s2)+dist(cow[j],s1)+dist(s1,s2)>m){
            G[i+n].push_back(j+n);
            G[j].push_back(i);
        }
    }
    for(int i = 1; i <= (n << 1); i++) if(!dfn[i]) tarjan(i);
    for(int i = 1; i <= n; i++) if(sccid[i]==sccid[i+n]) return false;
    return true;
}
int main(){
    while(scanf("%d %d %d",&n,&N,&M)!=EOF){
        scanf("%d %d %d %d",&s1.first,&s1.second,&s2.first,&s2.second);
        for(int i = 1; i <= n; i++) scanf("%d %d",&cow[i].first,&cow[i].second);
        for(int i = 1; i <= N; i++) scanf("%d %d",&emmm[i].first,&emmm[i].second);
        for(int i = 1; i <= M; i++) scanf("%d %d",&frd[i].first,&frd[i].second);
        int l = 1, r = 10000000;
        while(l<=r){
            int mid = (l + r) >> 1;
            if(_2sat(mid)) r = mid - 1;
            else l = mid + 1;
        }
        if(l==10000001) puts("-1");
        else printf("%d\n",l);
    }
    return 0;
}
```
