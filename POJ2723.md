2-sat,二分最深位置，然后对于每对🔑，选了一把就不能选另一把，对于每把锁，不选其中一把就必须要选另一把，建图判断是否可行
```cpp
#include<cstdio>
#include<cstring>
#include<iostream>
#include<set>
#include<map>
#include<vector>
#include<queue>
#include<string>
#include<algorithm>
#include<stack>
using namespace std;
const int MAXN = 1e4+7;
vector<int> G[MAXN];
int n,m,dfn[MAXN],low[MAXN],sccid[MAXN],instk[MAXN],idx,num;
stack<int> stk;
pair<int,int> key[MAXN], lock[MAXN];
void tarjan(int u){
    dfn[u] = low[u] = ++idx;
    instk[u] = 1;
    stk.push(u);
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(!dfn[v]){
            tarjan(v);
            low[u] = min(low[u],low[v]);
        }
        else{
            if(!instk[v]) continue;
            low[u] = min(low[u],dfn[v]);
        }
    }
    if(dfn[u]==low[u]){
        int tp;
        ++num;
        do{
            tp = stk.top();
            stk.pop();
            instk[tp] = false;
            sccid[tp] = num;
        }while(tp!=u);
    }
}
bool _2sat(int mid){
    int nn = n << 1;
    for(int i = 1; i <= (nn << 1); i++){
        G[i].clear();
        dfn[i] = low[i] = instk[i] = 0;
    }
    num = idx = 0;
    for(int i = 1; i <= n; i++){
        int x = key[i].first + 1, y = key[i].second + 1;
        G[x].push_back(y+nn);
        G[y].push_back(x+nn);
    }
    for(int i = 1; i <= mid; i++){
        int x = lock[i].first + 1, y = lock[i].second + 1;
        G[x+nn].push_back(y);
        G[y+nn].push_back(x);
    }
    for(int i = 1; i <= (nn << 1); i++) if(!dfn[i]) tarjan(i);
    for(int i = 1; i <= nn; i++) if(sccid[i]==sccid[i+nn]) return false;
    return true;
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF && n + m){
        for(int i = 1; i <= n; i++) scanf("%d %d",&key[i].first,&key[i].second);
        for(int i = 1; i <= m; i++) scanf("%d %d",&lock[i].first,&lock[i].second);
        int l = 0, r = m;
        while(l <= r){
            int mid = (l+r) >> 1;
            if(_2sat(mid)) l = mid + 1;
            else r = mid - 1;
        }
        printf("%d\n",r);
    }
    return 0;
}
```
