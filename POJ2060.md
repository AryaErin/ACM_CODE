初始情况下，假设每个乘客都需要一个出租车，现在一辆出租车接了每个乘客如果可以接上下一个乘客，那么就可以少用一辆出租车，所以就是要计算最多能连着接上几个
乘客，用总顾客数去减。
每个顾客向下一个可以接到的顾客连一条边，跑二分图匹配即可
```cpp
#include<cstring>
#include<cstdio>
#include<iostream>
#include<cmath>
#include<vector>
using namespace std;
const int MAXN = 555;
struct Graph{
    int tot, head[MAXN], to[MAXN*MAXN], nxt[MAXN*MAXN];
    void clear(){ tot = 0; memset(head,255,sizeof(head)); }
    void ADDEDGE(int u, int v){
        tot++;
        to[tot] = v;
        nxt[tot] = head[u];
        head[u] = tot;
    }
}G;
int n,tm[MAXN],match[MAXN];
bool vis[MAXN];
pair<int,int> s[MAXN],t[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = G.head[u]; ~i; i = G.nxt[i]){
        int v = G.to[i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int hungary(){
    int tot = 0;
    memset(match,255,sizeof(match));
    for(int i = 1; i <= n; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
void solve(){
    scanf("%d",&n);
    G.clear();
    for(int i = 1; i <= n; i++){
        int h, m;
        scanf("%d:%d %d %d %d %d",&h,&m,&s[i].first,&s[i].second,&t[i].first,&t[i].second);
        tm[i] = h*60+m;
    }
    for(int i = 1; i <= n; i++){
        int endtime = tm[i]+fabs(s[i].first-t[i].first)+fabs(s[i].second-t[i].second);
        for(int j = i + 1; j <= n; j++) {
            int starttime = endtime + fabs(t[i].first-s[j].first) + fabs(t[i].second-s[j].second);
            if(starttime<tm[j]) G.ADDEDGE(i,j);
        }
    }
    printf("%d\n",n-hungary());
}
int main(){
    int T;
    for(scanf("%d",&T); T; T--) solve();
    return 0;
}
```
