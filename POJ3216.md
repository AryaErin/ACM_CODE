void ____(){ ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0); }
const int MAXN = 222;
const int INF = 0x3f3f3f3f;
int n,m,match[MAXN],vis[MAXN],dist[MAXN][MAXN];
pair<int,pair<int,int> > work[MAXN];
vector<int> G[MAXN];
bool dfs(int u){
    vis[u] = true;
    for(int i = 0; i < (int)G[u].size(); i++){
        int v = G[u][i];
        if(match[v]==-1||(!vis[match[v]]&&dfs(match[v]))){
            match[v] = u;
            return true;
        }
    }
    return false;
}
int hungary(){
    memset(match,255,sizeof(match));
    int tot = 0;
    for(int i = 1; i <= m; i++){
        memset(vis,0,sizeof(vis));
        if(dfs(i)) tot++;
    }
    return tot;
}
int main(){
    while(scanf("%d %d",&n,&m)!=EOF&&n+m){
        for(int i = 1; i <= n; i++) for(int j = 1 ; j <= n; j++){
            scanf("%d",&dist[i][j]);
            if(dist[i][j]==-1) dist[i][j] = INF;
        }
        for(int k = 1; k <= n; k++) for(int i = 1; i <= n; i++) for(int j = 1; j <= n; j++){
            dist[i][j] = min(dist[i][j],dist[i][k]+dist[k][j]);
        }
        for(int i = 1; i <= m; i++) scanf("%d %d %d",&work[i].first,&work[i].second.first,&work[i].second.second);
        for(int i = 1; i <= m; i++) G[i].clear();
        for(int i = 1; i <= m; i++) for(int j = 1; j <= m; j++){
            if(work[i].second.first+work[i].second.second+dist[work[i].first][work[j].first]<=work[j].second.first){
                G[i].push_back(j);
            }
        }
        printf("%d\n",m-hungary());
    }
    return 0;
}
