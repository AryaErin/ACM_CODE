

删掉某条边(1,v)之后，必须还有边能够到达v且最短路不变
```cpp
//#pragma GCC optimize("O3")
//#pragma comment(linker, "/STACK:1024000000,1024000000")
#include<bits/stdc++.h>
using namespace std;
function<void(void)> ____ = [](){ios_base::sync_with_stdio(false); cin.tie(0); cout.tie(0);};
typedef long long int LL;
const int MAXN = 2e5+7;

int n, m, k;
vector<pair<int,LL> > G[MAXN];
LL dist[MAXN];
void Dijkstra(){
    memset(dist,0x3f,sizeof(dist));
    dist[1] = 0;
    priority_queue<pair<LL,int>,vector<pair<LL,int>>, greater<pair<LL,int>>> que;
    que.push(make_pair(dist[1],1));
    while(!que.empty()){
        auto p = que.top();
        que.pop();
        int u = p.second;
        LL w = p.first;
        if(dist[u]!=w) continue;
        for(auto e : G[u]){
            if(dist[u] + e.second < dist[e.first]){
                dist[e.first] = dist[u] + e.second;
                que.push(make_pair(dist[e.first],e.first));
            }
        }
    }
}
int main(){
    ____();
    cin >> n >> m >> k;
    vector<pair<pair<int,int>,int> > es;
    for(int i = 1; i <= m; i++){
        int u, v, w;
        cin >> u >> v >> w;
        es.push_back(make_pair(make_pair(u,v),w));
        G[u].push_back(make_pair(v,w));
        G[v].push_back(make_pair(u,w));
    }
    vector<pair<int,int>> dd;
    for(int i = 1; i <= k; i++){
        int v, w; cin >> v >> w;
        G[1].push_back(make_pair(v,w));
        dd.push_back(make_pair(v,w));
    }
    Dijkstra();
    vector<int> deg(n+1,0);
    for(auto &e : es){
        int u = e.first.first, v = e.first.second, w = e.second;
        if(dist[u]>dist[v]) swap(u,v);
        if(dist[u] + w == dist[v]) deg[v]++;
    }
    for(auto &e : dd) if(dist[e.first]==e.second) deg[e.first]++;
    int __count = 0;
    for(auto &e : dd){
        int v = e.first, w = e.second;
        if(dist[v]<w) __count++;
        else{
            if(deg[v]==1) continue;
            deg[v]--;
            __count++;
        }
    }
    cout << __count << endl;
    return 0;
}
```
